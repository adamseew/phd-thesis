    type=notation,
    name={$x_{i,j}$},

    type=notation,
    name={$\underline{x}_{i,j}$},
    description={lower bound of the parameter $x_{i,j}$}
}

\newglossaryentry{not:upbound}{
    type=notation,
    name={$\overline{x}_{i,j}$},
    description={upper bound of the parameter $x_{i,j}$}
}

\newglossaryentry{not:cardinal}{
    type=notation,
    name={$|x|$},
    description={cardinality of a set $x$}
}

\newglossaryentry{not:optimal}{
    type=notation,
    name={$x^*$},
    description={optimal with respect to a given cost}
}

\newglossaryentry{not:intdivision}{
    type=notation,
    name={$\lfloor x\rfloor$},
    description={the floor function or integer division of $x\in\mathbb{R}$, if $x$ is a set, the closest item in the set s.t. $x-\lfloor x\rfloor$ is positive}
}

\newglossaryentry{not:ceilfun}{
    type=notation,
    name={$\lceil x\rceil$},
    description={the ceiling function of $x\in\mathbb{R}$, if $x$ is a set, the closest item in the set s.t. $x-\lceil x\rceil$ is negative}
}

\newglossaryentry{not:edge}{
    type=notation,
    name={${}^{v_1}|_{v_2}$},
    description={edge connecting vertices $v_1$ and $v_2$}
}
